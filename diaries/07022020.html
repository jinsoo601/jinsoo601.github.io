        jinsoo.pickUpLaundry(cleanClothes);
    });
}

    MENU.SQUID_INK_PASTA,
    MENU.FRIED_CALAMARI,
    MENU.WATERMELON_MOJITO,
    MENU.PROSECCO,
    MENU.MARGARITA
]).then(food => {
    jinsoo.eatWithRoommates(food);
});

League.login(
    jinsoo.getLeagueId(),
    jinsoo.getLeaguePw()
).then(async (match) => {
    let victoryCount = 0;
    while (victoryCount < 3) {
        const matchResult = await match.joinAndPlay();
        if (matchResult === VICTORY) victoryCount++;

    League.logout();
});
