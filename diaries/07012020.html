import Person from "person";
import { SamsungTv, Laptop } from "entertainment";

const jinsoo = new Person("Jinsoo");

jinsoo.wakeup();
jinsoo.eat("grape");

const tv = new SamsungTv();
tv.on();
tv.selectPlatform("YouTube");
while (sally.isWorking()) {
    tv.playRandom()
}
tv.off();

const laptop = new Laptop();
laptop.start("League").then(async (league) => {
    let hasWon = false;
    while (!hasWon) {
        hasWon = await league.play(); // returns true if won, false if lost
    };
});
